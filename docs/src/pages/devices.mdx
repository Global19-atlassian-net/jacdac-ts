import { Grid } from '@material-ui/core';
import { Link } from "gatsby"
import { graphql } from "gatsby"

export const query = graphql`
  query {
    allDevicesJson {
      nodes {
        id
        name
      }
    }
  }
`

Known devices are specified in [https://github.com/microsoft/jacdac/tree/main/devices](https://github.com/microsoft/jacdac/tree/main/devices).

<ul>
    {props.data.allDevicesJson.nodes.map(node => <li>
        <Link key={node.id} to={`/devices/${node.id}`}>{node.name}</Link>
    </li>)}
</ul>
